        def copyAuxFiles
                Dir.chdir(@curDir)
                File.copy(@dtreeJSFile, @destDir)
                File.copy(@dtreeCSSFile, @destDir)

                @destDirImg = File.expand_path(@destDirImg)
                Dir.chdir(@dtreeImgDir)

                }
