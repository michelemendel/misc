    def tail
        puts "----- tail -f #{@name} -----\n\n"
        
        @session.session.open_channel do |channel|
            channel.on_data do |ch, data|
                puts data
            end
            channel.exec "tail -f #{@dir}#{@name}"
        end
        @session.session.loop
    end
